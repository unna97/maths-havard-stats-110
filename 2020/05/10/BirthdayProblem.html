import random
import matplotlib.pyplot as plt

n_party = 25
n_choices = 365
n_simulations = 100
a = [[random.choice(range(n_choices)) for i in range(n_party)] for j in range(n_simulations)]

x = [len(set(i)) == 25 for i in a]
plt.plot(range(100),x)
